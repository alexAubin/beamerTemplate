%%%%%%%%%%           Macros           %%%%%%%%%%%

\newcommand{\centered}[1]
{
	\begin{center}
		#1
	\end{center}
}

\newcommand{\itemList}[2]
{
    \textbf{#1}
	\begin{itemize}
		#2
	\end{itemize}
}

\newcommand{\slide}[1]
{
	\begin{frame}
		#1
	\end{frame}
}

\newcommand{\slideCentered}[1]
{
	\begin{frame}
	\begin{center}
		#1
	\end{center}
	\end{frame}
}

\newcommand{\eqn}[1]
{
	\begin{eqnarray}
		#1
	\end{eqnarray}
}


\newcommand{\printOutline}
{
	\section*{Outline}
	\subsection*{Outline}

	\slide
	{
		\setcounter{tocdepth}{1}
		\tableofcontents
		\setcounter{tocdepth}{2}
	}
}

\newcommand{\printTitle}
{
	{
		\setbeamertemplate{footline}{}
		\slide
		{
			\titlepage
		}
	}
}

\newcommand{\ignoreNextCounting}
{
	\addtocounter{framenumber}{-1}
}

%%%%%%%%%%%%%%%%%%%%%
% Images management
%%%%%%%%%%%%%%%%%%%%%

\newcommand{\imgw}[2]
{
	\includegraphics[width=#1\textwidth]{#2}
}
\newcommand{\imgh}[2]
{
    \includegraphics[height=#1\textwidth ]{#2}
}

\newcommand{\twoImgs}[2]
{
	\begin{tabular}{cc}
		\includegraphics[width=0.5\textwidth]{#1} &
		\includegraphics[width=0.5\textwidth]{#2}
	\end{tabular}
}
\newcommand{\twoImgsWithCaptions}[4]
{
    \begin{tabular}{cc}
        \includegraphics[width=0.5\textwidth]{#1} &
        \includegraphics[width=0.5\textwidth]{#3}\\
        #2 & #4
    \end{tabular}
}

%%%%%%%%%%%%%%%%%%%%%

\newcommand{\twoColumns}[2]
{
    \begin{minipage}{0.49\textwidth}
        #1
    \end{minipage}
    \begin{minipage}{0.49\textwidth}
        #2
    \end{minipage}
}

%%%%%%%%%%%%%%%%%%%%%
% For backup slides
%%%%%%%%%%%%%%%%%%%%%

\newcommand{\backupBegin}
{
	\newcounter{n_backupslides}
	\setcounter{n_backupslides}{\value{framenumber}}
}
\newcommand{\backupEnd}
{
	\addtocounter{n_backupslides}{-\value{framenumber}}
	\addtocounter{framenumber}{\value{n_backupslides}}
}
