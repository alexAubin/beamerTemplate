{

	}

\newcommand{\blockw}[2] {\begin{minipage}{#1\textwidth}#2\end{minipage}}
\newcommand{\imgw}[2] {\includegraphics[width=#1\textwidth]{#2}}
\newcommand{\pdfw}[2] {\includegraphics[width=#1\textwidth,angle=-90]{#2}}
\newcommand{\imgh}[2] {\includegraphics[height=#1\textwidth]{#2}}
\newcommand{\pdfh}[2] {\includegraphics[height=#1\textwidth,angle=-90]{#2}}

\newcommand{\twoImgs}[2]
{
	\begin{tabular}{cc}
        \imgw{0.5}{#1} &
        \imgw{0.5}{#2}
	\end{tabular}
}
\newcommand{\twoImgsWithCaptions}[4]
{
    \begin{tabular}{cc}
        #1 & #3 \\
        \imgw{0.5}{#2} &
        \imgw{0.5}{#4}
    \end{tabular}
}

\newcommand{\twoPlotsWithCaptions}[4]
{
    \hspace*{-0.08\textwidth}
    \begin{tabular}{rl}
        \blockw{0.55}{\centered{\large{#1}}} & 
        \blockw{0.55}{\centered{\large{#3}}} \\
        \pdfw{0.46}{#2} &
        \pdfw{0.46}{#4} 
    \end{tabular}
}

\setlength{\TPHorizModule}{\paperwidth}

\newcommand{\noteblock}[5]
{
    {
       \begin{textblock}{#1}[0,0](#2,#3)
       {
           \color{#4}
           \setlength{\fboxsep}{1pt}
           \setlength{\fboxrule}{1pt}
           \fbox{\setlength{\fboxrule}{0.5pt}\setlength{\fboxsep}{2.5pt}\fbox{#5}}
       }
       \end{textblock}
    }
}
